var sketchProc = function(processingInstance) {

with (processingInstance) {

size(400, 400);

frameRate(30);

// ProgramCodeGoesHere

//Shooting Star position


var xPos = 50;

var yPos = 200;

var rPos = 12;


//moon position


var xMoon = 200;

var yMoon = 100;

var rMoon = 70;


draw = function() {

background(29, 40, 115);

//draw moon

fill(242, 242, 242);

ellipse(xMoon, yMoon, rMoon, rMoon);

//moon craters

noStroke();

fill(209, 205, 205);

ellipse(xMoon + 15, yMoon + 20, rMoon - 55, rMoon - 55 );

ellipse(xMoon - 10, yMoon - 20, rMoon - 65, rMoon - 65 );

ellipse(xMoon - 17, yMoon + 20, rMoon - 50, rMoon - 55 );

ellipse(xMoon + 15, yMoon - 15, rMoon - 60, rMoon - 60 );

//Star

fill(255, 242, 0);

ellipse(xPos, yPos, rPos, rPos);

//draw buildings

fill(0, 0, 0);

rect(200, 200, 75, 200); //building 1

fill(255, 255, 0);

rect(215, 220, 10, 10); // windows

rect(250, 250, 10, 10);

rect(215, 300, 10, 10);

rect(250, 220, 10, 10);

fill(0, 0, 0);

rect(290, 150, 80, 250); //building 2

fill(255, 255, 0);

rect(300, 190, 10, 10); // windows

rect(350, 210, 10, 10);

rect(300, 240, 10, 10);

//draw tree

fill(56, 27, 27);

rect(100, 380, 10, 20);

fill(10, 112, 22);

triangle(80, 380, 105, 350, 130, 380);

triangle(85, 360, 105, 330, 125, 360);

//move star up/right

xPos += 1;

yPos -= 1;

rPos *= 499/500;

//move moon slow left

xMoon -= 0.5;

yMoon -= 0.1;

};


//End function!

}};


// Get the canvas that Processing-js will use

var canvas = document.getElementById("mycanvas");

// Pass the function sketchProc (defined in myCode.js) to Processing's constructor.

var processingInstance = new Processing(canvas, sketchProc);

</script>
